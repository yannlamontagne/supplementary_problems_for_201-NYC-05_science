\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
    Is there a set of four vectors in \( \Re^3 \) such that any three 
    form a linearly independent set?
\end{Exercise}

\begin{Answer}
      Yes; here is one.
      \begin{equation*}
         \set{\colvec{1 \\ 0 \\ 0},
              \colvec{0 \\ 1 \\ 0},
              \colvec{0 \\ 0 \\ 1},
              \colvec{1 \\ 1 \\ 1} }
      \end{equation*}  
\end{Answer}
