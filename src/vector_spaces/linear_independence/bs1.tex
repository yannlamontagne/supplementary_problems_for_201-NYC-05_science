\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
Let $A=\begin{mat}
1 & 2\\
0 & 1
\end{mat}$.
Determine whether the vectors $\trans{A}$, $A^{-1}$ and $A^2$ are linearly independent in $\matspace_\nbyn{2}$.
\end{Exercise}

\begin{Answer}
$\trans{A}$, $A^{-1}$ and $A^2$ are linearly independent.
\end{Answer}
