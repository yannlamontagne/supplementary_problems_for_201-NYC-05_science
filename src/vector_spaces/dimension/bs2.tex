\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
Let $V_n=\set{A\in\matspace_{\nbyn{n}} \suchthat \trans{A}=-A}$.  Find a basis and the dimension of $V_3$.
\end{Exercise}

\begin{Answer}
$\set{
\begin{mat}
0 & 1 & 0\\
-1 & 0 & 0\\
0 & 0 & 0
\end{mat},\; 
\begin{mat}
0 & 0 & 1\\
0 & 0 & 0\\
-1 & 0 & 0
\end{mat},\;
\begin{mat}
0 & 0 & 0\\
0 & 0 & 1\\
0 & -1 & 0
\end{mat}
}$ is a basis of $V_3$.  $\text{dim}\left(V_3\right)=3$.
\end{Answer}
