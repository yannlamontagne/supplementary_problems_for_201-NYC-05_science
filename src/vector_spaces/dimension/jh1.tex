\begin{Exercise}[
name={},

difficulty=0,
origin={\cite{JH}}]
Find a basis for, and the dimension of, \(  \polyspace_2 \).
\end{Exercise}

\begin{Answer}
One basis is \( \sequence{1,x,x^2} \), and so
the dimension is three.
\end{Answer}
