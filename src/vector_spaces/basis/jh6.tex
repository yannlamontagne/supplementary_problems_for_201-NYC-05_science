\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
    Find a basis for \( \matspace_{\nbyn{2}} \),
    the space of \( \nbyn{2} \) matrices.
\end{Exercise}

\begin{Answer}
      There are many bases.
      This is a natural one.
      \begin{equation*}
        \sequence{
           \begin{mat}[r]
             1  &0  \\
             0  &0
           \end{mat},
           \begin{mat}[r]
             0  &1  \\
             0  &0
           \end{mat},
           \begin{mat}[r]
             0  &0  \\
             1  &0
           \end{mat},
           \begin{mat}[r]
             0  &0  \\
             0  &1
           \end{mat}  }
      \end{equation*} 
\end{Answer}
