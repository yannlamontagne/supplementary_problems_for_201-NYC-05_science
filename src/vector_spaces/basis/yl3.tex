\begin{Exercise}[

difficulty=0,
origin={\cite{YL}}]
Given\\
\begin{align*}
W=& \left\{a_0+a_1x+a_2x^2+a_3x^3\suchthat  a_0+2a_1+3a_2+4a_3=0\right. \\ 
&\left. \text{ and }a_1+a_2+a_3=0\right\}
\end{align*} 
a subspace of $\polyspace_3$.  Find a basis for $W$.
\end{Exercise}

\begin{Answer}
$\set{-1-x+x^2, -2-x+x^3}$
\end{Answer}
