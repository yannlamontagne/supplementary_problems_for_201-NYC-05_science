\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Determine whether\\
$W=\set{\begin{mat}
a & b\\
c & d
\end{mat}
\suchthat
(ab)^2=(cd)^2}$
is a subspace of $\matspace_{2\times 2}$.

\end{Exercise}

\begin{Answer}
Not a subspace since it is not closed under vector addition.
\end{Answer}
