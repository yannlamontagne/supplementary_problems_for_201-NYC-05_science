\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
Determine whether the $W = \set{(x,\; y,\; z) \in \R^3 \suchthat xy = 0\text{ and }yz = 0}$ is a subspace of $\R^3$.
\end{Exercise}

\begin{Answer}
$W$ is not a subspace since it is not closed under addition.
\end{Answer}
