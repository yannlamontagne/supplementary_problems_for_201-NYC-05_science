\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Determine whether\\
$W=\left\{a_0+a_1x+a_2x^2\;\mid\; a_0a_1a_2=0\right\}$
is a subspace of  \( \polyspace_2 \).
\end{Exercise}

\begin{Answer}
Not a subspace since it is not closed under vector addition.
\end{Answer}
