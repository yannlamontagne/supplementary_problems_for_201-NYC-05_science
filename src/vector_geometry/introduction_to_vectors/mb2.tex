\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{SM}}]
Determine whether the three points $P(1,-1,0)$, $Q(2,4,1)$ and $R(-1,-11,-2)$ are collinear.
\end{Exercise}

\begin{Answer}
Since $\vec{PR}=-2\vec{PQ}$, the points $P$, $Q$ and $R$ are colinear.
\end{Answer}
