\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
    Decide if the two vectors are equal.
    \Question the vector from \( (5,3) \) to \( (6,2) \) and the vector
        from \( (1,-2) \) to \( (1,1) \)
    \Question the vector from \( (2,1,1) \) to \( (3,0,4) \) and the vector
        from \( (5,1,4) \) to \( (6,0,7) \)


\end{Exercise}

\begin{Answer}
\Question No, they are different.
          \begin{equation*}
            \colvec[r]{1 \\ -1}
            \qquad
            \colvec[r]{0 \\ 3}
          \end{equation*}
\Question Yes, they are the same.
          \begin{equation*}
            \colvec[r]{1 \\ -1 \\ 3}
          \end{equation*}
\end{Answer}
