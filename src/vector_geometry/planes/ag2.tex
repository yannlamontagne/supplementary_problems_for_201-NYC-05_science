\begin{Exercise}[
name={},

Let $\Pi$ be the plane in $\R^3$ with normal vector $$\vec{n}=
\begin{mat}
\cos\theta\\
\cos\theta+\sin\theta\\
\sin\theta
\end{mat}
\;.$$ Find all the values of $\theta$ in the interval $[0,2\pi)$ such that $\Pi$ is parallel to the plane of vector equation
$$
\begin{mat}
x\\
y\\
z
\end{mat}
=
\begin{mat}
1\\
0\\
0
\end{mat}
+r
\begin{mat}
1\\
-1\\
1
\end{mat}
+t
\begin{mat}
1\\
-1\\
-2
\end{mat}
$$.

\end{Exercise}

\begin{Answer}
$0,\;\pi$
\end{Answer}
