\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{GHC}}]
Show, using the definition of the cross product, that $\vec u\times\vec u=\vec 0$.

\end{Exercise}

\begin{Answer}
With $\vec u = (u_1,u_2,u_3)$, we have
\begin{align*}
\vec u\times\vec u &= (u_2u_3-u_3u_2,-(u_1u_3-u_3u_1),u_1u_2-u_2u_1) \\
		&= (0,0,0)\\
		&=\vec 0.
\end{align*}

\end{Answer}
