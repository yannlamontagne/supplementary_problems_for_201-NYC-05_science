\begin{Exercise}[

difficulty=0,
origin={\cite{MH}}]
Let $A=\begin{mat}a&b&c\\d&e&f\\g&h&i\end{mat}$ such that $\det A=10$. Let $\vec u$, $\vec v$ and $\vec w$ be the columns of $A$. Find $\vec u\cdot(\vec v\times\vec w)$.
\end{Exercise}

\begin{Answer}
$10$
\end{Answer}
