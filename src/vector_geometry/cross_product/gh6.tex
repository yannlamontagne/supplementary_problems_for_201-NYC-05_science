Find the area of the triangle with the given vertices.
\Question $(0,0,0)$, $(1,3,-1)$ and $(2,1,1)$.
\Question $(5,2,-1)$, $(3,6,2)$ and $(1,0,4)$.
\Question $(1,1)$, $(1,3)$ and $(2,2)$.
\Question $(3,1)$, $(1,2)$ and $(4,3)$.
\end{Exercise}

\Question $5\sqrt{2}/2$
\Question $3\sqrt{30}/2$
\Question $1$
\Question $5/2$
\end{Answer}
