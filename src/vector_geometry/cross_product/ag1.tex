\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{AG}}]
A parallelepiped has the vectors
$$
\vec{u}=
\begin{mat}
a\\
2\\
1
\end{mat}
,\quad
\mathbf{v}=
\begin{mat}
0\\
a\\
-2
\end{mat}
,\quad
\vec{w}=
\begin{mat}
1\\
1\\
1
\end{mat}
$$
as its edges. Find the values of $a$ so that it will have volume $V=8$.
\end{Exercise}

\begin{Answer}
$-4,\;3$
\end{Answer}
