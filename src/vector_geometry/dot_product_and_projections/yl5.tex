\begin{Answer}
\emph{hint:} A tangent to a circle is perpendicular to the radius at the point at which the tangent and circle intersect.
\begin{multicols}{2}
\Question $C(4,\;0)$
\Question $B(1,\;3)$
\Question $(x-4)^2+y^2=18$
\EndCurrentQuestion
\end{multicols}
\end{Answer}
