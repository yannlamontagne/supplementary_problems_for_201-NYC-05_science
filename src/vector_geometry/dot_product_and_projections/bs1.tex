\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
If $\vec{u}=(0, 1, 1)$ and $\vec{v}=(p, 4, p)$ then find the parameter $p$ such that the angle between $\vec{u}$ and $\vec{v}$ is $\pi/3$.
\end{Exercise}

\begin{Answer}
$p=-1$
\end{Answer}
