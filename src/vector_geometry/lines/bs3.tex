\begin{Exercise}[

Find the distance between the origin and the line passing through the points $A(-2, 0, -1)$ and $B(-2, 1, 0)$.
\end{Exercise}
\begin{Answer}
$\frac{3}{\sqrt{2}}$
\end{Answer}
