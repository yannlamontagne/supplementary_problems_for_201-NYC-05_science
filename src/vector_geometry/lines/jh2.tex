\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
Does \( (1,0,2,1) \) lie on the line through \( (-2,1,1,0) \) and \( (5,10,-1,4) \)?
\end{Exercise}

\begin{Answer}
      That line is this set.
      \begin{equation*}
        \set{\colvec[r]{-2 \\ 1 \\ 1 \\ 0}
             +\colvec[r]{7 \\ 9 \\ -2 \\ 4}t \suchthat t\in\Re }
      \end{equation*}
      Note that this system
      \begin{equation*}
        \begin{linsys}{2}
          -2  &+  &7t  &=  &1  \\
           1  &+  &9t  &=  &0  \\
           1  &-  &2t  &=  &2  \\
           0  &+  &4t  &=  &1  
        \end{linsys}
      \end{equation*}
      has no solution.
      Thus the given point is not in the line.  
\end{Answer}
