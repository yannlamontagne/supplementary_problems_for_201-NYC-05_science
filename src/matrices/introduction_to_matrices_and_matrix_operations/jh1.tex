\begin{Exercise}[

Find the indicated entry of the following matrix.
\begin{equation*}
A=\begin{mat}
1  &3  &1  \\
2  &-1 &4
\end{mat}
\end{equation*}
\begin{multicols}{4}
\Question \( a_{2,1} \) 
\Question \( a_{1,2} \)  
\Question \( a_{2,2} \)
\Question \( a_{3,1} \) 
\EndCurrentQuestion
\end{multicols}

\end{Exercise}

\begin{Answer}
\Question $2$
\Question $3$
\Question $-1$
\Question Not defined.
\end{Answer}
