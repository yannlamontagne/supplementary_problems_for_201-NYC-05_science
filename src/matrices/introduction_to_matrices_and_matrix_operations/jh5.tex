\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
Find the product of this matrix with its transpose.
\begin{equation*}
\begin{mat}
\cos\theta  &-\sin\theta  \\
\sin\theta  &\cos\theta
\end{mat}
\end{equation*}
\end{Exercise}

\begin{Answer}
The product is the identity matrix (recall that
$\cos^2\theta+\sin^2\theta =1$).
\end{Answer}
