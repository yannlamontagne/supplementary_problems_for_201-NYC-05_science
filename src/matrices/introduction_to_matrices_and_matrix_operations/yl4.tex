\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Solve for all $a,\;b,\;c,\;d$ such that
\[
\begin{mat}
3a  +  3b  +  7c - 3d &
2a  +  3b  +  3c + d \\
4a  +  17c -  2d  &
9a  +  6b  +  27c - 4d 
\end{mat}
=
\begin{mat}
1 & 2\\
3 & 6
\end{mat}
\]
\end{Exercise}

\begin{Answer}
$(a,\;b,\;c,\;d)=(-29+60t,\; 13-\frac{79}{3}t,\; 7-14t,\;t)$ where $t\in\Re$.
\end{Answer}
