\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{KK}}]
Show that the main diagonal of every skew symmetric matrix consists of only zeros.
\end{Exercise}

\begin{Answer}
f $A$ is skew-symmetric then $A=-A^{T}.$ It follows that $a_{ii}=-a_{ii}$ and so each $a_{ii}=0$.
\end{Answer}
