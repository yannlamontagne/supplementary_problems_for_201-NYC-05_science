\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{KK}}]
 Let $A=\begin{mat}
-1 & -1 \\
3 & 3
\end{mat}$. Find all $2\times 2$ matrices, $B$ such that $AB=0.$ 
\end{Exercise}

\begin{Answer}
\begin{eqnarray*}
\begin{mat}
-1 & -1 \\
3 & 3
\end{mat}
\begin{mat}
x & y \\
z & w
\end{mat} &=&
\begin{mat}
-x-z & -w-y \\
3x+3z & 3w+3y
\end{mat} \\
&=&
\begin{mat}
0 & 0 \\
0 & 0
\end{mat}
\end{eqnarray*}
Solution is: $ w=-y,x=-z $ so the
matrices are of the form $
\begin{mat}
x & y \\
-x & -y
\end{mat}$
\end{Answer}
