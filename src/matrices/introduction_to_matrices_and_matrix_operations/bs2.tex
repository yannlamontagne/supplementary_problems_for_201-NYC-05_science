\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
Find all matrices $A$ such that
\[
\begin{mat}
1 & 1 & 1\\
0 & 1 & 1
\end{mat}A=
\begin{mat}
1 & 0\\
0 & 1
\end{mat}
\]
\end{Exercise}

\begin{Answer}
$
\begin{mat}
1 & -1 \\
-s & 1-t\\
s & t
\end{mat}
$ where $s,\;t\in\mathbb{R}$
\end{Answer}
