\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{KK}}]
A matrix $A$ is called {\em idempotent \em} if $A^{2}=A.$  
Show that the following matrix is idempotent.
\[
A=
\begin{mat}
2 & 0 & 2 \\
1 & 1 & 2 \\
-1 & 0 & -1
\end{mat}
\]
\end{Exercise}

\begin{Answer}
Evaluate $A^2=AA$.
\end{Answer}
