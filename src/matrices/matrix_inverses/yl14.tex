\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Show that if $A$ is invertible then its inverse is unique.
\end{Exercise}

\begin{Answer}
Suppose that that there exists two inverse then show that they are equal using the fact that they are both the inverse of $A$.
\end{Answer}
