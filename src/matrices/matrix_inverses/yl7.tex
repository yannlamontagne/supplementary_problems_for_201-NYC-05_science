\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Show that a matrix $A$ which satisfy $A^3+3A^2+A+I=0$ is invertible and express its inverse in terms of $A$ and the identity.
\end{Exercise}

\begin{Answer}
Show that $A^{-1}=-A^2-3A -I$.
\end{Answer}
