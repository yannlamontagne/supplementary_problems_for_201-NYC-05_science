\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
    In real number algebra, there are exactly two numbers, $1$ and $-1$, 
    that are their own multiplicative inverse.
    Does \( H^2=I \) have exactly two solutions for \( \nbyn{2} \)
    matrices?
\end{Exercise}
\begin{Answer}
      Here are four solutions to \( H^2=I \).
      \begin{equation*}
        \begin{mat}[r]
          \pm 1  &0  \\
          0      &\pm 1
        \end{mat}
      \end{equation*}   
\end{Answer}
