\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{KK}}]
Suppose $AB=AC$ and $A$ is an invertible $n\times n$ matrix. Does it
follow that $B=C$? Explain. 
\end{Exercise}

\begin{Answer}
Yes $B=C$. Multiply $AB = AC$ on the left by $A^{-1}$.
\end{Answer}
