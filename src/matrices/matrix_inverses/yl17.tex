\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Solve for $X$ given that it satisfies
\[
\Big(2A+X^T\Big)^{-1}=I
\]
where
\[
A=
\begin{mat}
1 & 3\\
1 & 2
\end{mat}\;
\]

\end{Exercise}

\begin{Answer}
$
X=
\begin{mat}
-1 & -2\\
-6 & -3
\end{mat}
$
\end{Answer}
