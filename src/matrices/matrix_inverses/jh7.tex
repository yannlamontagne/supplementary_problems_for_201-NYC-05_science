\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
    Assume that \( H \) is invertible and that \( HG \) is the zero matrix.
    Show that \( G \) is a zero matrix.
\end{Exercise}
\begin{Answer}
      The associativity of matrix multiplication gives
      \( H^{-1}(HG)=H^{-1}Z=Z \) and also
      \( H^{-1}(HG)=(H^{-1}H)G=IG=G \).
\end{Answer}
