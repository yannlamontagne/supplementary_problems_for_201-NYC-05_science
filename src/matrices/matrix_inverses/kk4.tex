\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{KK}}]
Show $\left( AB\right) ^{-1}=B^{-1}A^{-1}$ by verifying that 
\begin{equation*}
AB\left(
B^{-1}A^{-1}\right) =I\text{ and }B^{-1}A^{-1}\left( AB\right) =I
\end{equation*}
\end{Exercise}

\begin{Answer}
$\left( AB\right)
B^{-1}A^{-1}=A\left( BB^{-1}\right) A^{-1}=AA^{-1}=I$ 
$B^{-1}A^{-1}\left(
AB\right) =B^{-1}\left( A^{-1}A\right) B=B^{-1}IB=B^{-1}B=I$
\end{Answer}
