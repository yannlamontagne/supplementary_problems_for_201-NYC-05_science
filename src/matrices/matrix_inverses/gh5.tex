\begin{Exercise}[
name={},

difficulty=0,
origin={\cite{GH}}]
Prove or disprove: If $A$ and $B$ are $\nbyn{2}$ invertible matrices then $A+B$ is an invertible matrix.

\end{Exercise}
\begin{Answer}
Disprove: $A=\begin{mat}1&0\\0&1\end{mat}\;B=\begin{mat}-1&0\\0&-1\end{mat}$.
\end{Answer}
