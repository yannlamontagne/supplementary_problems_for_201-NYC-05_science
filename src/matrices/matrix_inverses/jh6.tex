\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
Prove: If $T$ is an invertible matrix and $k$ is a natural number then 
\( (T^k)^{-1}=(T^{-1})^k \). 
\end{Exercise}
\begin{Answer}
     \( T^k(T^{-1})^k=(TT\cdots T)\cdot (T^{-1}T^{-1}\cdots T^{-1})
         =T^{k-1}(TT^{-1})(T^{-1})^{k-1}=\dots=I \). 
\end{Answer}
