\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Given an $n\times n$ identity $I$ and $p(x)=a_0+a_1x+a_2x^2+\ldots+a_mx^m$ an 
arbitrary polynomial of degree $m$.  Find the condition for which $p(I)$ is invertible.
\end{Exercise}

\begin{Answer}
Invertible if $a_0+a_1+a_2+\ldots+a_m\neq 0$.
\end{Answer}
