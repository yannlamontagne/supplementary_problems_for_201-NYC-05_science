\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{AG}}]
A matrix $A$ is called \emph{nilpotent} if $A^k=0$ for some positive integer $k$. Show that if $A$ is nilpotent for a certain $k$ then $(I-A)^{-1}=I+A+A^2+\cdots+A^k$.
\end{Exercise}

\begin{Answer}
Show that $(I-A)(I+A+A^2+\cdots+A^k)=I$
\end{Answer}
