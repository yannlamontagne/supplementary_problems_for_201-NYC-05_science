\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
    Prove that if \( H \) is invertible then
    the inverse commutes with a matrix \( GH^{-1}=H^{-1}G \) 
    if and only if $H$ itself commutes with that matrix \( GH=HG \).
\end{Exercise}
\begin{Answer}
       Multiply both sides of the first equation by $H$.
\end{Answer}
