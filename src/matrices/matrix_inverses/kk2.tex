\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{KK}}]
Give an example of a matrix $A$ such that $A^{2}=I$, $A\neq I$
and $A\neq -I.$ 
\end{Exercise}
\begin{Answer}
$A =
\begin{mat}
1 & 0 & 0 \\
0 & -1 & 0 \\
0 & 0 & 1
\end{mat}$
\end{Answer}
