\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{MH}}]
An \emph{involutory matrix} is a matrix $A$ such that $A^2=I$.
Show that if $A$ is an involutory matrix, then $A$ is invertible and find $A^{-1}$.
\end{Exercise}

\begin{Answer}
Since $AA=I$ it follows that $A^{-1}=A$.
\end{Answer}
