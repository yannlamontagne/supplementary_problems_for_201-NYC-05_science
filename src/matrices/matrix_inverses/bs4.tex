\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
Find all matrices $A$ such that
\[
A^{-1}\begin{mat}
1 & 0 & 0\\
0 & 1 & 0\\
7 & 0 & 1
\end{mat}
=
\left(
3I_3
-2A
\right)^{-1}
\]
\end{Exercise}

\begin{Answer}
$
A=\begin{mat}
1 & 0 & 0\\
0 & 1 & 0\\
\frac{7}{3} & 0 & 1
\end{mat}
$
\end{Answer}
