\[
A=
\begin{mat}
1 & 2 & 3\\
0 & 1 & 2\\
0 & 0 & 3
\end{mat}.
\]

\Question Find $A^{-1}$.
\Question Using $A^{-1}$ solve $Ax=b$ where
\[
x=\begin{mat}x_1\\ x_2\\ x_3\end{mat}\text{ and }b=\begin{mat}1\\ -2\\ 1\end{mat}.
\]


\end{Exercise}
