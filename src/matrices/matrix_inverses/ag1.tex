\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{AG}}]
Solve for $A$ the following equation:
$$
(3A^T-I)^{-1}C-D=0
$$
where $C=\begin{pmatrix}
7&-1\\
6&-1
\end{pmatrix}
$
and $D=\begin{pmatrix}
5&4\\
1&1
\end{pmatrix}
$.
\end{Exercise}

\begin{Answer}
$
A=\begin{mat}
3 & \frac{7}{3}\\
-11 & -\frac{28}{3}
\end{mat}
$
\end{Answer}
