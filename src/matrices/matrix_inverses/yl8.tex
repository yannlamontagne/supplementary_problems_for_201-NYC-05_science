\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Find a formula for $\trace(A^{-1})$ if $A=\begin{bmatrix}a&b\\c&d\end{bmatrix}$ is invertible.
\end{Exercise}

\begin{Answer}
$\trace{A^{-1}}=\frac{a+d}{ad-bc}$
\end{Answer}
