\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{OV}}]
Find matrix $A$, if
\[
\trans{\left(2\trans{A}-3\begin{mat}1 & -1\\ 0 & 4\end{mat}\right)}=\begin{mat}1 & 2\\ 3 & 3\end{mat}
\]
\end{Exercise}

\begin{Answer}
$A=\begin{mat}2 & 1\\ 0 & 8\end{mat}$
\end{Answer}
