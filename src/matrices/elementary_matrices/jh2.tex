\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
Find
\Question a $\nbyn{3}$ matrix that, acting from the left,
         swaps rows one and two.
\Question a $\nbyn{2}$ matrix that, acting from the right,
         swaps column one and two.
\end{Exercise}

\begin{Answer}
\Question This matrix swaps row one and row three.
          \begin{equation*}
            \begin{mat}
              0  &1 &0 \\
              1  &0 &0 \\
              0  &0 &1
            \end{mat}
            \begin{mat}
              a &b &c \\
              d &e &f \\
              g &h &i
            \end{mat}
            =
            \begin{mat}
              d &e &f \\
              a &b &c \\
              g &h &i
            \end{mat}         
          \end{equation*}
\Question This matrix swaps column one and two.
          \begin{equation*}
            \begin{mat}
              a  &b  \\
              c  &d
            \end{mat}
            \begin{mat}
              0  &1  \\
              1  &0   
            \end{mat}
            =
            \begin{mat}
              b  &a  \\
              d  &c
            \end{mat}
          \end{equation*}
\end{Answer}
