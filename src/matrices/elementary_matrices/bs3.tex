\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
Find all matrices $A$ such that 
\[
A(3I-2A)^{-1}=\begin{mat}
1 & 0 & 3\\
0 & 1 & 0\\
0 & 0 & 1
\end{mat}
\]
\end{Exercise}
\begin{Answer}
$A=\begin{mat}
1 & 0 & 1\\
0 & 1 & 0\\
0 & 0 & 1
\end{mat}$

\end{Answer}
