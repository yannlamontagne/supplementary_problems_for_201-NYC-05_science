\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Express
\[
A = 
\begin{mat}
1 & 0 & 0\\
1 & 0 & 1\\
0 & 2 & 1
\end{mat}
\]
as a product of 4 elementary matrices.
\end{Exercise}

\begin{Answer}
\[
A=
\begin{mat}
1 & 0 & 0\\
1 & 1 & 0\\
0 & 0 & 1
\end{mat}
\begin{mat}
1 & 0 & 0\\
0 & 0 & 1\\
0 & 1 & 0
\end{mat}
\begin{mat}
1 & 0 & 0\\
0 & 1 & 1\\
0 & 0 & 1
\end{mat}
\begin{mat}
1 & 0 & 0\\
0 & 2 & 0\\
0 & 0 & 1
\end{mat}
\]
Note: The answer is not unique.
\end{Answer}
