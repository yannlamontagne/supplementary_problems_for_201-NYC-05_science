\Question Maximize $Z=3x+y$ subject to the constraints
\begin{align*}
2x-y &\leq 60\\
x+y&\leq50.
\end{align*}
\Question Maximize $Z=2x+y+3z$ subject to the constraints 
\begin{align*}
2x-y+z&\leq100\\
x+y+2z&\leq70.
\end{align*}
\Question Maximize $Z=2x+y+3z$ subject to the constraints 
\begin{align*}
2x-y-z&\leq10\\ 
x+y+2z&\leq60\\
x-y-z&\leq10.
\end{align*}
\Question Maximize $Z=4x+3y+3z$ subject to the constraints 
\begin{align*}
y-z&\leq10\\ 
2x-y+2z&\leq60\\ 
2x-y-z&\leq10\\ 
x+y+2z&\leq30.
\end{align*}
\Question Maximize $Z=2x-y+4z+2w$ subject to the constraints 
\begin{align*}
y+z+w&\leq30\\
2x-y+2z-w&\leq60\\ 
2x-y-z-w&\leq10.
\end{align*}
\end{Exercise}
