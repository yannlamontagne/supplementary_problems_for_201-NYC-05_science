\Question Minimize $Z = x+y$ subject to the constraints
\begin{align*}
x+y  & \geq 2\\
3x+y & \geq 4.
\end{align*}
\Question Minimize $Z = 2x+y+z$ subject to the constraints
\begin{align*}
x+z & \geq 2\\
2x+y+z & \geq 3.
\end{align*}
\Question Minimize $Z = 2x+y+3z$ subject to the constraints
\begin{align*}
x+4z   & \geq 2\\
2x+y+z & \geq 3\\ 
x+y+z & \geq 1.
\end{align*}
\Question Minimize $Z = x+y+z$ subject to the constraints
\begin{align*}
x+y   & \geq 50\\
y+z & \geq 50\\ 
x+z & \geq 10.
\end{align*}
\Question Minimize $Z=3x+y-2z$ subject to the constraints
\begin{align*}
2x-y&\geq60\\
x+y-2z&\geq50\\
x-y-z&\geq10.
\end{align*}

\end{Exercise}

\begin{Answer}
\Question $Z = 2$; $x = 1$, $y = 1$ from the final tableau
\[
\begin{mat}
x & y & s_1 & s_2 & -Z & \\
0 & 1 & -3/2 & 1/2 & 0 & 1\\
1 & 0 & 1/2 & -1/2 & 0 & 1\\
0 & 0 & 1 & 0 & 1 & -2
\end{mat}
\]
