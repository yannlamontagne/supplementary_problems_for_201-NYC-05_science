\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{GH}}]
Solve the system of linear equations using substitution, comparison and/or elimination.
\begin{multicols}{2}
\Question
$
\begin{linsys}{2}
x&+&y&=&-1\\
2x&-&3y&=&8\\
\end{linsys}
$

\Question
$
\begin{linsys}{2}
2x&-&3y&=&3\\
3x&+&6y&=&8\\
\end{linsys}
$

\Question
$
\begin{linsys}{3}
x&-&y&+&z&=&1\\
2x&+&6y&-&z&=&-4\\
4x&-&5y&+&2z&=&0\\
\end{linsys}
$

\Question
$
\begin{linsys}{3}
x&+&y&-&z&=&1\\
2x&+&y&&&=&2\\
&&y&+&2z&=&0\\
\end{linsys}
$
\EndCurrentQuestion
\end{multicols}
\end{Exercise}

\begin{Answer}
\Question $x=1$, $y=-2$
\Question $x=2$, $y=\frac13$
\Question $x=-1$, $y=0$, and $z=2$.
\Question $x=1$, $y=0$, and $z=0$.
\end{Answer}
