\begin{Exercise}[name={},
title={}, 
origin={\cite{OV}},
counter=Exercise]
Determine conditions on the $b_i$'s, if any, in order to guarantee that the linear system is consistent.
\[
\begin{linsys}{3}
x &   &   & + & z & = & b_1\\
x & + & y & + & 2z & = & b_2\\
2x & + & y & + & 3z & = & b_3\\
\end{linsys}
\]
\end{Exercise}

\begin{Answer}
$b_3=b_1+b_2$
\end{Answer}
