\begin{Exercise}[name={},
title={}, 
origin={\cite{BS}},
counter=Exercise]
Consider the system
\[
\begin{array}{rrrrrrrr}
kx & + & y &+ &kz & =& 1\\
x   & + & y & + & z & =&  1\\
(2-k)x  & + & (2-k)y&+ &z & =&  1\\
kx  & + & y&+ &kz & =&  k^2\\
\end{array}
\]
Find the value(s) of $k$, if any, such that the system has:
\Question no solutions,
\Question a unique solution, 
\Question infiniely many solutions.
\end{Exercise}

\begin{Answer}
\begin{multicols}{3}
\Question $k\neq\pm 1$
\Question $k=-1$
\Question $k=1$
\EndCurrentQuestion
\end{multicols}
\end{Answer}
