\begin{Exercise}[name={},
title={}, 
origin={\cite{OV}},
counter=Exercise]
Find $k$ for which the system
\[
\begin{linsys}{3}
x_1 & + & kx_2 & & & = & 0\\
&&(k-1)x_2 & + & x_3 & = & 0\\
x_1 &  & & + & (k+2)x_3 & = & 0\\
\end{linsys}
\]
\Question has infinitely many solutions.
\Question has only the trivial solution.
\end{Exercise}

\begin{Answer}
\Question $k=-1\pm\sqrt{3}$
\Question $k\neq-1\pm\sqrt{3}$
\end{Answer}
