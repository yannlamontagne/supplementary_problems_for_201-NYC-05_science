\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{SZ}}]
Find at least two different row echelon forms for the matrix
\[
\begin{mat}
1 & 2 & 3 \\ 
4 & 12 & 8 
\end{mat}
\]
\end{Exercise}

\begin{Answer}
$\begin{mat}1 & 2 & 3\\ 0 & 1 & -1\end{mat}$ and $\begin{mat}1 & 3 & 2\\0&1&-1 \end{mat}$
\end{Answer}
