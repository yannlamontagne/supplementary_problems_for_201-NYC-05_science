\begin{Exercise}[name={},

Given the following matrix:
\[
A=
\begin{mat}
3 & 1 & 2 & 4\\
1 & 0 & 2 & 0\\
3 & 1 & 0 & 2
\end{mat}.
\]
Solve the linear system whose coefficient matrix is the matrix $A$ and which has a particular solution $(1,\; 0,\; -1,\; 0)$.
\end{Exercise}
