\begin{Exercise}[
name={},

difficulty=0,
origin={\cite{YL}}]
Prove or disprove: If $\det(A^2)=\det(A)$ then $A^2=A$.
\end{Exercise}

\begin{Answer}
Disprove using the counter example $A=\begin{mat}
0 & 1\\
0 & 0
\end{mat}
$.
\end{Answer}
