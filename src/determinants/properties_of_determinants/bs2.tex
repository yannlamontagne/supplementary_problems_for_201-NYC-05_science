\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
If $A$ and $B$ are $\nbyn{3}$ matrices such that $AB^T = I$ and $\det(A) = 2$, find $\det(A^2B)$.
\end{Exercise}

\begin{Answer}
$\frac{1}{2}$
\end{Answer}
