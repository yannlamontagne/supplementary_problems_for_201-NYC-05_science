\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Given
\[
A=\begin{mat}
a & 0 & 0 & 0\\
0 & b & 0 & 0\\
0 & 0 & c & 0\\
0 & 0 & 0 & d
\end{mat}
\text{ and }
B=\begin{mat}
0 & 0 & \frac3a & \frac3a\\
0 & 0 & 0 & \frac2b\\
0 & \frac1c & 0 & \frac1c\\
\frac4d & 0 & 0 & \frac4d
\end{mat}.
\]
If $\det(B)=5$ then determine $\det(A)$.
\end{Exercise}

\begin{Answer}
$\det(A)=\frac{-24}{5}$
\end{Answer}
