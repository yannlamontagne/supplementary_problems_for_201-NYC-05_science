\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
      Which real numbers \( x \) make this matrix singular?
      \begin{equation*}
         \begin{mat}
            12-x  &4  \\
            8    &8-x
         \end{mat}
      \end{equation*}

\end{Exercise}

\begin{Answer}
        This equation
         \begin{equation*}
           0=
           \det(
             \begin{mat}
                12-x  &4  \\
                8    &8-x
             \end{mat}
           )
           =64-20x+x^2
           =(x-16)(x-4) 
       \end{equation*}
       has roots \( x=16 \) and \( x=4 \).  
\end{Answer}
