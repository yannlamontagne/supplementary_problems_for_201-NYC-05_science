\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{JH}}]
Prove: If $S$ and $T$ are $\nbyn{n}$ matrix then $\det(TS)=\det(ST)$.  
\end{Exercise}

\begin{Answer}
\( \det(TS)=\det(T)\cdot\det(S)=\det(S)\cdot\det(T)=\det(ST) \).  
\end{Answer}
