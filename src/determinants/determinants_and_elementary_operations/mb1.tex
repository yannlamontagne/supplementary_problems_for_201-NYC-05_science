\begin{Exercise}[
name={},
title={},
difficulty=0,
origin={\cite{MB}}]
Compute the determinant of the matrix
$$ 
A= \begin{mat}
\phantom{-}1 & \phantom{-}2 & \phantom{-}0  & -3  \\ 
0 & 4 & 2 & 1 \\ 
0 & 2 & -1 & -1 \\
2 & 3 & 1 & 0
\end{mat}$$
by combining row operations and cofactor expansions.
\end{Exercise}

\begin{Answer}
$
\detop(A) = -41
$
\end{Answer}
