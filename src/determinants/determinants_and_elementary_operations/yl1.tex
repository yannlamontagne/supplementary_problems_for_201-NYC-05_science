\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Consider
\[
A
=
\begin{mat}
a & d & g\\
b & e & h\\
c & f & k
\end{mat}
\text{ and }
B
=
\begin{mat}
3d & 3e & 3f\\
a+2d & b+2e & c+2f\\
4g & 4h & 4k
\end{mat}.
\]
If $\detop(B)=5$ then determine $\detop(A)$.


\end{Exercise}

\begin{Answer}
$
\detop(A) = -\frac{5}{12}
$
\end{Answer}
