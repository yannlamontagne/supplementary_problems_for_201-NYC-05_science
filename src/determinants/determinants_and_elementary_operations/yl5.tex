\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Using elementary operations show that
\[
-sr\begin{vmatrix}
a & b\\
c & d
\end{vmatrix}
=
\begin{vmatrix}
sb+2d & rsa+2rc\\
d & rc
\end{vmatrix}
\]
\end{Exercise}

\begin{Answer}
Use a combination of elementary row and column operation to get from one side of the equation to the other.
\end{Answer}
