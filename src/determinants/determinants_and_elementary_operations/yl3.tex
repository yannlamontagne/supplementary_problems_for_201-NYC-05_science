\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Consider
\[
A=
\begin{mat}
a & b\\
c & d 
\end{mat},\;
B=
\begin{mat}
c & d \\
-7a-5c & -7b-5d 
\end{mat},\;
C=
\begin{mat}
b & 2a-3b \\
d & 2c-3d
\end{mat}.
\]
\Question If $\det(B)=5$ then determine $\det(A)$.
\Question If $\det(C)=5$ then determine $\det(A)$.

\end{Exercise}

\begin{Answer}
\begin{multicols}{2}
\Question $\det(A)=\frac{2}{7}$
\Question $\det(A)=-\frac{5}{2}$
\EndCurrentQuestion
\end{multicols}
\end{Answer}
