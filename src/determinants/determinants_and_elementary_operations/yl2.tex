\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
If
$
A=\begin{mat}
a&0&b\\
0&c&0\\
d&0&e
\end{mat}
$
and $\det(2A^T)=16$ then evaluate
$
\begin{detmat}2a & 4a & 6a+12b\\ 0 & 8c & 10c\\ 2d & 4d & 6d+12e\end{detmat}.
$
\end{Exercise}

\begin{Answer}
$
384
$
\end{Answer}
