\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{GH}}]
A matrix $M$ and $\det(M)$ are given. Matrices $A$, $B$ and $C$ are obtained by performing operations on $M$. Determine the determinants of $A$, $B$ and $C$ and indicate the operations used to obtain $A$, $B$ and $C$.
\begin{multicols}{2}
\Question $M=\begin{mat}9&7&8\\  1&3&7\\  6&3&3 \end{mat},\\ \det(M)=-41,\\\\ A=\begin{mat}18&14&16\\  1&3&7\\  6&3&3 \end{mat},\\ B=\begin{mat}9&7&8\\  1&3&7\\  96&73&83 \end{mat},\\ C=\begin{mat}9&1&6\\  7&3&3\\  8&7&3 \end{mat}$.
\Question $M=\begin{mat} 0&3&5\\  3&1&0\\  -2&-4&-1 \end{mat},\\ \det(M)=45,\\\\ A=\begin{mat}0&3&5\\  -2&-4&-1
\\  3&1&0 \end{mat},\\ B=\begin{mat}0&3&5\\  3&1&0\\  8&16&4 \end{mat},
\\ C=\begin{mat}3&4&5\\  3&1&0\\  -2&-4&-1 \end{mat}$.
\Question $M=\begin{mat}5&1&5\\  4&0&2\\  0&0&4 \end{mat},\\ \det(M)=-16,\\\\ A=\begin{mat}0&0&4\\  5&1&5\\  4&0&2 \end{mat},\\ B=\begin{mat}-5&-1&-5\\  -4&0&-2\\  0&0&4 \end{mat},\\ C=\begin{mat}15&3&15\\  12&0&6\\  0&0&12 \end{mat}$.
\Question $M=\begin{mat}5&4&0\\  7&9&3\\  1&3&9 \end{mat},\\ \det(M)=120,\\\\ A=\begin{mat}1&3&9\\  7&9&3\\  5&4&0 \end{mat},\\ B=\begin{mat}5&4&0\\  14&18&6\\  3&9&27 \end{mat},\\ C=\begin{mat}-5&-4&0\\  -7&-9&-3\\  -1&-3&-9 \end{mat}$.
\EndCurrentQuestion
\end{multicols}

\end{Exercise}
\begin{Answer}
\Question $\det(A)= 90$; $2R_1\rightarrow R_1$.\\ 
$\det(B)=45$; $10R_1+R_3\rightarrow R_3$.\\
$\det(C)=45 $; $C=A^T$.
\Question $\det(A)=41$; $R_2\leftrightarrow R_3$.\\
 $\det(B)=164$; $-4R_3\rightarrow R_3$.\\
 $\det(C)= -41 $; $R_2+R_1\rightarrow R_1$.
\Question $\det(A)= -16$; $R_1\leftrightarrow R_2$ then $R_1\leftrightarrow R_3$.\\
$\det(B)= -16$; $-R_1\rightarrow R_1$ and $-R_2\rightarrow R_2$.\\
$\det(C)= -432$; $C=3M$.
\Question $\det(A)= -120$; $R_1\leftrightarrow R_2$ then $R_1\leftrightarrow R_3$ then $R_2\leftrightarrow R_3$.\\
$\det(B)= 720$; $2R_2\rightarrow R_2$ and $3R_3\rightarrow R_3$.\\
$\det(C)= -120$; $C=-M$.
\end{Answer}
