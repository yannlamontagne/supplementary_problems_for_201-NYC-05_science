\begin{Exercise}[
name={Vandermonde's determinant},
title={}, 
difficulty=0,
origin={\cite{JH}}]
Prove:
\begin{equation*}
\det\left(
         \begin{mat}
             1    &1   &1    \\
             a    &b   &c    \\
             a^2  &b^2 &c^2
         \end{mat}
         \right)
=(b-a)(c-a)(c-b)
\end{equation*}
\end{Exercise}

\begin{Answer}
Hint: Use elementary operations to bring the matrix under triangular form.
\end{Answer}
