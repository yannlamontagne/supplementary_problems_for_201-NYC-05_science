\Question Find a formula for the adjoint of a \( \nbyn{2} \) matrix.
\Question Use the above to derive the formula for the inverse of a \( \nbyn{2} \) matrix. 
\end{Exercise}

\begin{Answer}
\Question $\begin{mat}
             T_{1,1}  &T_{2,1}  \\
             T_{1,2}  &T_{2,2}
            \end{mat}
            =\begin{mat}
            \begin{vmat}
              t_{2,2}
            \end{vmat}
           &-\begin{vmat}
               t_{1,2}
             \end{vmat}       \\
           -\begin{vmat}
               t_{2,1}
            \end{vmat}
           &\begin{vmat}
              t_{1,1} 
            \end{vmat}
          \end{mat}
          =\begin{mat}
             t_{2,2}  &-t_{1,2}  \\
            -t_{2,1} &t_{1,1}
           \end{mat}$
\Question $(1/t_{1,1}t_{2,2}-t_{1,2}t_{2,1})
          \begin{mat}
             t_{2,2}  &-t_{1,2}  \\
            -t_{2,1} &t_{1,1}
           \end{mat}$


\end{Answer}
