\begin{Exercise}[

origin={\cite{MB}}]
Use Cramer's rule to solve this system.
$$  
\begin{linsys}{3}
x_1 & + & 2x_2  & = & -1 \\
-x_1 & +  & -4x_2 & = & 2  
\end{linsys}.  
$$
\end{Exercise}

\begin{Answer}
$x_1=0$, $x_2=\frac{-1}{2}$
\end{Answer}
