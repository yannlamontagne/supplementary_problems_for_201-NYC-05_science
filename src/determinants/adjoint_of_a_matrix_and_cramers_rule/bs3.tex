\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
If $A$ and $B$ are invertible matrices of the same size show that
\[
\text{adj}\left(AB\right)=\text{adj}\left(B\right)\text{adj}\left(A\right)
\].
\end{Exercise}

\begin{Answer}
Hint: Use $\text{adj}\left(A\right)=\left(\det A\right)A^{-1}$.
\end{Answer}
