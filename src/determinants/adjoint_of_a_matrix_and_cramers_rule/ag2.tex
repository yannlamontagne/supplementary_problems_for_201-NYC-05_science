\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{AG}}]
Let $A$ and $B$ be two $\nbyn{3}$ matrices with $\det(A)=2$ and $\det(B)=5$. Find $\det\left(A^3\det(B)\trans{A}(\text{adj}(A))^4\right)$.
\end{Exercise}

\begin{Answer}
$5^32^{12}$
\end{Answer}
