\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
If $A$ is a $\nbyn{3}$ matrix such that $\det(A)=2$ find
\[
\det\left(\text{adj}\left(2\text{adj}\left(A^{-1}\right)\right)\right)
\].
\end{Exercise}

\begin{Answer}
$4$
\end{Answer}
