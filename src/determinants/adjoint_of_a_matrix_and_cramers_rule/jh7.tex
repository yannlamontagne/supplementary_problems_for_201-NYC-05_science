\begin{Exercise}[

Find the adjoint of the following matrices.
\begin{multicols}{2}
\Question \( \begin{mat}[r]
                 2   &1  &4  \\
                -1   &0  &2  \\
                 1   &0  &1
               \end{mat}  \)
\Question \( \begin{mat}[r]
                 3  &-1  \\
                 2  &4
               \end{mat}  \)
\Question \( \begin{mat}[r]
                 1   &1  \\
                 5   &0
               \end{mat}  \)
\Question \( \begin{mat}[r]
                 1   &4  &3  \\
                -1   &0  &3  \\
                 1   &8  &9
               \end{mat}  \)
\Question $\begin{mat}[r]
        2  &1  &0  &0  \\
        1  &2  &1  &0  \\
        0  &1  &2  &1  \\
        0  &0  &1  &2
      \end{mat}$
\EndCurrentQuestion
\end{multicols}
\end{Exercise}
