\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
Given $A$, an $n\times n$ matrix such that $\det(A)=9$ and
\[
A^3A^T=3A^{-1}\text{adj}(A)
\]
find $n$.
\end{Exercise}

\begin{Answer}
$4$
\end{Answer}
