\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{MB}}]
Consider two $\nbyn{4}$ matrices $A$ and $B$, with $\det(A) = -2$ and $\det(B) = 3$. Find the determinant of $M$, knowing that $\det(2\trans{B}MA^{-1}B)=\det(\adj(A)A^2B)$.
\end{Exercise}

\begin{Answer}
$\det(M)=\frac{4}{3}$
\end{Answer}
