\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
Solve for $x$ where
\[
\begin{vmat}
\sin x & \cos x\\
-\cos x & \sin x
\end{vmat}
=
\begin{vmat}
1 & 1 & 1\\
e^x & 1 & e^x\\
1 & e^x & 0
\end{vmat}
\]
\end{Exercise}
\begin{Answer}
$\ln 2$
\end{Answer}
