\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{BS}}]
Given
$
A=
\begin{bmatrix}
x^2 & 1 & x\\
3 & 1 & 2\\
x & -1 & -1
\end{bmatrix}
$
find the value(s) of $x$, if any, such that $\det(A)=0$.
\end{Exercise}
\begin{Answer}
$3$
\end{Answer}
